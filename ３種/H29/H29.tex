\section{問５}

\begin{align*}
P_{n} = \sqrt{3} V_{n} I_{n} 
\end{align*}
より
\begin{align*}
 I_{n} &= \dfrac{P_{n}}{\sqrt{3} V_{n}}\\
&=\dfrac{10 \times 10^{6}}{6.6 \times 10^{3}} \\
&\fallingdotseq 875.8 \mathrm{[A]}
\end{align*}
百分率インピーダンス$\%Z_{s}\mathrm{[\%]}$は三相短絡電流を$I_{s} \mathrm{[A]}$とすると
\begin{align*}
\% Z_{n} = \frac{I_{n}}{I_{s}} \times 100 \mathrm{[\%]}
\end{align*}
より
\begin{align*}
I_{s} &= \frac{I_{n}}{\% Z } \times 100 \mathrm{[\%]}\\
& =\frac{875.8}{80} \\
& \fallingdotseq 1094.8 \mathrm{[A]}
\end{align*}
短絡電流$I_{f}\mathrm{[A]}$と三相短絡電流は比例関係にあるので，
\begin{align*}
700:50&=1094.8:I_{f}\\
I_{f}&=\frac{1094.8 \times 50}{700}\\
&\fallingdotseq  78.1 \mathrm{[A]}
\end{align*}
